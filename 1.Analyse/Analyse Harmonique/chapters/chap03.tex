\section{Convergence Theory in \(L^{2}\) Space} % L2 空间中的收敛理论
\begin{theorem}{Carleson-Hunt Theorem}
    If \(f \in L^{2}(\mathbb{T})\), then the Fourier series of \(f\) converges to \(f\) almost everywhere.
\end{theorem}
